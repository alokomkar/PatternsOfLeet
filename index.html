The Two Pointers Mechanism

🧠 Secret Clues to Spot This Pattern:

• Input is a linear array/string.
• The elements are explicitly Sorted, or sorting them optimizes space to $O(1)$.
• You are searching for a pair or comparison boundaries from both ends.

Two Sum II - Sorted InputEASY

"Find two numbers in a sorted array that sum up to a target."

fun twoSum(numbers: IntArray, target: Int): IntArray {
    var left = 0; var right = numbers.size - 1
    while (left < right) {
        val sum = numbers[left] + numbers[right]
        if (sum == target) return intArrayOf(left + 1, right + 1)
        else if (sum < target) left++ else right--
    }
    return intArrayOf()
}
Container With Most WaterMEDIUM

"Find two lines that form a container holding the most water."

fun maxArea(height: IntArray): Int {
    var left = 0; var right = height.size - 1; var maxWater = 0
    while (left < right) {
        val width = right - left
        val currentHeight = minOf(height[left], height[right])
        maxWater = maxOf(maxWater, width * currentHeight)
        if (height[left] < height[right]) left++ else right--
    }
    return maxWater
}

The Sliding Window Mechanism

🧠 Secret Clues to Spot This Pattern:

• Requires continuous segments (subarrays or substrings).
• Seeks boundaries defining phrases like "longest", "shortest", or "size K".

Maximum Average Subarray IEASY

"Find a contiguous subarray of fixed size K with the highest average."

fun findMaxAverage(nums: IntArray, k: Int): Double {
    var sum = 0.0; for (i in 0 until k) sum += nums[i]
    var maxSum = sum
    for (i in k until nums.size) {
        sum += nums[i] - nums[i - k]
        maxSum = maxOf(maxSum, sum)
    }
    return maxSum / k
}
Longest Substring Without Repeating CharsMEDIUM
fun lengthOfLongestSubstring(s: String): Int {
    var left = 0; var maxLength = 0; val seen = HashSet()
    for (right in s.indices) {
        while (seen.contains(s[right])) { seen.remove(s[left]); left++ }
        seen.add(s[right]); maxLength = maxOf(maxLength, right - left + 1)
    }
    return maxLength
}

The Prefix Sum Optimization

🧠 Secret Clues to Spot This Pattern:

• Multiple range total queries over an immutable array.
• Subarray summation checks where sliding windows fail due to negative values.

Range Sum Query - ImmutableEASY
class NumArray(nums: IntArray) {
    private val prefix = IntArray(nums.size + 1)
    init { for (i in nums.indices) prefix[i + 1] = prefix[i] + nums[i] }
    fun sumRange(left: Int, right: Int) = prefix[right + 1] - prefix[left]
}
Subarray Sum Equals KMEDIUM
fun subarraySum(nums: IntArray, k: Int): Int {
    var count = 0; var currentSum = 0; val prefixMap = HashMap(); prefixMap[0] = 1
    for (num in nums) {
        currentSum += num
        if (prefixMap.containsKey(currentSum - k)) count += prefixMap[currentSum - k]!!
        prefixMap[currentSum] = prefixMap.getOrDefault(currentSum, 0) + 1
    }
    return count
}

Cycle Detection Framework

⚠️ Google Edge-Case Constraint:

Array implementations require a guaranteed duplicate entry. Without a loop trap, pointers will overrun array indices and throw a runtime exception.

Linked List Cycle CheckEASY
fun hasCycle(head: ListNode?): Boolean {
    var slow = head; var fast = head
    while (fast?.next != null) {
        slow = slow?.next; fast = fast.next?.next
        if (slow == fast) return true
    }
    return false
}
Find the Duplicate NumberMEDIUM
fun findDuplicate(nums: IntArray): Int {
    var slow = nums[0]; var fast = nums[0]
    do { slow = nums[slow]; fast = nums[nums[fast]] } while (slow != fast)
    var p1 = nums[0]; var p2 = slow
    while (p1 != p2) { p1 = nums[p1]; p2 = nums[p2] }
    return p1
}

The Monotonic Stack Engine

Next Greater Element IEASY
fun nextGreaterElement(queryArray: IntArray, searchArray: IntArray): IntArray {
    val answerMap = HashMap(); val waitingRoom = Stack()
    for (num in searchArray) {
        while (waitingRoom.isNotEmpty() && waitingRoom.peek() < num) answerMap[waitingRoom.pop()] = num
        waitingRoom.push(num)
    }
    return IntArray(queryArray.size) { answerMap.getOrDefault(queryArray[it], -1) }
}
Daily TemperaturesMEDIUM
fun dailyTemperatures(temperatures: IntArray): IntArray {
    val result = IntArray(temperatures.size); val stack = Stack()
    for (i in temperatures.indices) {
        while (stack.isNotEmpty() && temperatures[stack.peek()] < temperatures[i]) {
            val oldIdx = stack.pop(); result[oldIdx] = i - oldIdx
        }
        stack.push(i)
    }
    return result
}

State Machine Backtracking

Subsets (Power Set)EASY
fun subsets(nums: IntArray): List> {
    val res = mutableListOf>()
    fun generate(idx: Int, path: MutableList) {
        if (idx == nums.size) { res.add(ArrayList(path)); return }
        path.add(nums[idx]); generate(idx + 1, path) // Include
        path.removeAt(path.size - 1); generate(idx + 1, path) // Exclude
    }
    generate(0, mutableListOf()); return res
}
Permutations CollectionMEDIUM
fun permute(nums: IntArray): List> {
    val res = mutableListOf>()
    fun backtrack(path: MutableList) {
        if (path.size == nums.size) { res.add(ArrayList(path)); return }
        for (num in nums) {
            if (path.contains(num)) continue
            path.add(num); backtrack(path); path.removeAt(path.size - 1)
        }
    }
    backtrack(mutableListOf()); return res
}
